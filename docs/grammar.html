LL(1) Language Grammar

START -> <ASPECT>+
ASPECT -> "aspect" <ID> '{' <POINTCUT>* <ADVICE>* '}'
POINTCUT -> "pointcut" <ID> '=' <JOINPOINT> <JOINPOINTLIST>* ';'
ADVICE -> "advice" <ADVICETYPE> <PARAMLIST> ':' <JOINPOINT> <JOINPOINTLIST>* <CODEBLOCK>
PARAMLIST -> '('[<VARS>]')'
JOINPOINT -> <FJOINPOINT> | <VJOINPOINT> | <ID>
FJOINPOINT -> <FJPTYPE>'('<SIGNATURE><PARAMLIST>')'<ARGS>*
VJOINPOINT -> <VJPTYPE>'('<SIGNATURE>')'<ARGS>*
JOINPOINTLIST -> '|' <JOINPOINT>
SIGNATURE -> <ID>'.'<ID>
ARGS -> "&&" <ID> '('<ARGVAL>')'
CODEBLOCK -> <DELIM> <NONDELIM>* <ENDDELIM>
VARS -> '$'<ID><VARLIST>*
VAR_LIST -> ','<VAR>
VAR -> '$'<ID>
ID -> <LETTER><ALPHA>*
ARGVAL -> '"'<CHAR>'"' | <DIGIT>*
ADVICETYPE -> "before" | "after" | "around"
FJPTYPE -> "exec" | "execr"
VJPTYPE -> "set" | "get"
LETTER -> [A-Za-z]
DIGIT -> [0-9]
ALPHA -> <LETTER>|<DIGIT>
DELIM -> "{%"
ENDDELIM -> "%}"
NONDELIM -> ([^%]|[%][^}])*
CHAR -> .*
